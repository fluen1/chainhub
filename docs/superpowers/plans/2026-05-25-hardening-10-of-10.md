- [ ] **Step 3: Implement the rate limiter**

Model after existing `src/lib/ai/rate-limit.ts` (token-bucket pattern), but keyed on email with a sliding window:

```typescript
// src/lib/auth/login-rate-limit.ts
const MAX_ATTEMPTS = 5
const WINDOW_MS = 15 * 60 * 1000 // 15 min

interface LoginAttempt {
  count: number
  firstAttemptAt: number
}

const attempts = new Map<string, LoginAttempt>()

export function checkLoginRateLimit(email: string): { allowed: boolean; retryAfterMs?: number } {
  const key = email.trim().toLowerCase()
  const now = Date.now()
  const entry = attempts.get(key)

  if (!entry || now - entry.firstAttemptAt > WINDOW_MS) {
    attempts.set(key, { count: 1, firstAttemptAt: now })
    return { allowed: true }
  }

  if (entry.count < MAX_ATTEMPTS) {
    entry.count++
    return { allowed: true }
  }

  const retryAfterMs = WINDOW_MS - (now - entry.firstAttemptAt)
  return { allowed: false, retryAfterMs }
}

export function resetLoginRateLimiter(): void {
  attempts.clear()
}
```

```typescript
import { parsePaginationParams } from '@/lib/pagination'

interface TaskPaginationParams {
  page?: number
  pageSize?: number
  search?: string
  status?: string
  priority?: string
  sort?: string
  sortDir?: 'asc' | 'desc'
  assignedToMe?: boolean
}

export async function getTasksPaginated(params: TaskPaginationParams) {
  const session = await auth()
  if (!session) return { rows: [], totalCount: 0, page: 1, pageSize: 20 }

  const orgId = session.user.organizationId
  const userId = session.user.id
  const { page, skip, take } = parsePaginationParams(params.page, params.pageSize)

  const where: Prisma.TaskWhereInput = {
    organization_id: orgId,
    deleted_at: null,
    ...(params.search && {
      OR: [
        { title: { contains: params.search, mode: 'insensitive' } },
        { description: { contains: params.search, mode: 'insensitive' } },
      ],
    }),
    ...(params.status && { status: params.status as TaskStatus }),
    ...(params.priority && { priority: params.priority as TaskPriority }),
    ...(params.assignedToMe && { assigned_to_id: userId }),
  }

  const orderBy = params.sort
    ? { [params.sort]: params.sortDir ?? 'asc' }
    : { created_at: 'desc' as const }

  const [rows, totalCount] = await Promise.all([
    prisma.task.findMany({
      where,
      orderBy,
      skip,
      take,
      include: {
        assigned_to: { select: { id: true, name: true } },
        company: { select: { id: true, name: true } },
      },
    }),
    prisma.task.count({ where }),
  ])

  return { rows, totalCount, page, pageSize: take }
}
```

- [ ] **Step 3: Implement `getPersonsPaginated`**

Follow same pattern as Task 2.1 but for Person model. Include the RBAC filter (currently on page.tsx lines 39-42) as a `WHERE` clause — read page.tsx to see the exact filter condition and replicate it in the Prisma query. Key: the filter checks which companies the user can access, then filters persons by their company relations.

```typescript
import { parsePaginationParams } from '@/lib/pagination'
import { getAccessibleCompanyIds } from '@/lib/permissions'

export async function getPersonsPaginated(params: {
  page?: number
  pageSize?: number
  search?: string
  role?: string
}) {
  const session = await auth()
  if (!session) return { rows: [], totalCount: 0, page: 1, pageSize: 20 }

  const orgId = session.user.organizationId
  const userId = session.user.id
  const { page, skip, take } = parsePaginationParams(params.page, params.pageSize)

  // Move the RBAC filter from page.tsx into the WHERE clause
  const accessibleCompanyIds = await getAccessibleCompanyIds(userId, orgId)

  const where: Prisma.PersonWhereInput = {
    organization_id: orgId,
    deleted_at: null,
    company_persons: {
      some: { company_id: { in: accessibleCompanyIds } },
    },
    ...(params.search && {
      OR: [
        { first_name: { contains: params.search, mode: 'insensitive' } },
        { last_name: { contains: params.search, mode: 'insensitive' } },
        { email: { contains: params.search, mode: 'insensitive' } },
      ],
    }),
  }
```

```typescript
import { ErrorBoundaryUI } from '@/components/ui/b/ErrorBoundaryUI'

interface ErrorBoundaryPageProps {
  error: Error & { digest?: string }
  reset: () => void
  page: string
}

export function ErrorBoundaryPage({ error, reset, page }: ErrorBoundaryPageProps) {
  useEffect(() => {
    console.error(error)
    Sentry.captureException(error, { tags: { page } })
  }, [error, page])

  return <ErrorBoundaryUI error={error} reset={reset} />
}
```
